Overview

Provides methods to query properties of the CPU in the computer. This module's methods will be mixed into Hardware::CPU on MacOS X.

Several of these methods use info spewed out by sysctl. Look in <mach/machine.h> for decoding info.

def bits
  return @bits if defined? @bits

  is_64_bit = sysctl_bool("hw.cpu64bit_capable")
  @bits ||= is_64_bit ? 64 : 32
end

def family
  if type == :intel
    @intel_family ||= `/usr/sbin/sysctl -n hw.cpufamily`.to_i
    case @intel_family
    when 0x73d67300 # Yonah: Core Solo/Duo
      :core
    when 0x426f69ef # Merom: Core 2 Duo
      :core2
    when 0x78ea4fbc # Penryn
      :penryn
    when 0x6b5a4cd2 # Nehalem
      :nehalem
    when 0x573B5EEC # Arrandale
      :arrandale
    when 0x5490B78C # Sandy Bridge
      :sandybridge
    when 0x1F65E835 # Ivy Bridge
      :ivybridge
    else
      :dunno
    end
  elsif type == :ppc
    @ppc_family ||= `/usr/sbin/sysctl -n hw.cpusubtype`.to_i
    case @ppc_family
    when 9
      :g3  # PowerPC 750
    when 10
      :g4  # PowerPC 7400
    when 11
      :g4e # PowerPC 7450
    when 100
      :g5  # PowerPC 970
    else
      :dunno
    end
  end
end

def type
  @type ||= `/usr/sbin/sysctl -n hw.cputype`.to_i
  case @type
  when 7
    :intel
  when 18
    :ppc
  else
    :dunno
  end
end
